def createWordsVec(VocabList,inX):

def classify(inX,p0v,p1v,pclass1):
    p1=sum(inX*p1v)+log(pclass1)
    p0=sum(inX*p0v)+log(1-pclass1)
    if(p1>p0):
        return 1
    else:
        return 0
